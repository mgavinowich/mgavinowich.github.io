264 	/**
265 	 * Performs the search to get all experiments that match the specified
266 	 * criteria
267 	 * 
268 	 * @return a String for the navigation views indicating that the same view
269 	 *         should be presented
270 	 */
271 	public String search() {
272 
273 		ExperimentCRUDService crudService = ExperimentManagerEnvironment.getInstance().getServiceLocator()
274 				.getExperimentCRUDService();
275 
276 		logger.debug("Searching experiments");
277 
278 		try {
279 			List<ExperimentInfo> experimentsRead = crudService.matchExperiments(FacesUtils
280 					.getPortletPreference(PropertiesConstants.EXPERIMENT_TYPE_PREF_NAME), this.experimentName,
281 					this.experimentDescription, this.experimentAuthor, this.startDate, this.endDate);
282 
283 			logger.debug("Sorting experiments found");
284 
285 			List<ExperimentInfo> sortedExperiments = new ArrayList<ExperimentInfo>(experimentsRead);
286 
287 			Collections.sort(sortedExperiments, new Comparator<ExperimentInfo>() {
288 
289 				@Override
290 				public int compare(ExperimentInfo o1, ExperimentInfo o2) {
291 					return (o1.getName().compareToIgnoreCase(o2.getName()));
292 				}
293 
294 			});
295 
296 			this.experimentsFound = sortedExperiments;
297 		}
298 		catch (ExperimentDataManipulationException e) {
299 			MessageUtils.addMessageToContext(EXPERIMENT_STORAGE_ACCESS_PROBLEM);
300 
301 			logger.error("A problem accesing storage was encountered when trying to search experiments", e);
302 		}
303 
304 		this.resetDataScrollerToFirstPage();
305 
306 		logger.debug("Done searching experiments");
307 
308 		return STAY_IN_SAME_VIEW;
309 	}

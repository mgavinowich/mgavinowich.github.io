205 	/**
206 	 * If this instance does not reference any experiment, loads it from the
207 	 * corresponding request parameter
208 	 */
209 	private void lazyLoadExperimentFromRequestParameter() {
210 
211 		logger.debug("Checking if the experiment is already loaded in this bean");
212 
213 		if (this.innerExperiment == null) {
214 
215 			logger.debug("Experiment is not loaded, look up experiment name in request parameters");
216 
217 			String experimentName = FacesUtils.getRequestParameter(EXPERIMENT_NAME);
218 			if (experimentName == null) {
219 				MessageUtils.handleInternalErrorCondition(REQUEST_PARAM_NOT_FOUND, experimentName);
220 			}
221 			try {
222 				this.loadInnerExperiment(experimentName);
223 			}
224 			catch (ExperimentNotFoundException e) {
225 				MessageUtils.addMessageToContext(EXPERIMENT_NOT_FOUND, experimentName);
226 
227 				logger.error("The experiment " + experimentName
228 						+ " was not found when trying to load it in this backing bean", e);
229 			}
230 			catch (ExperimentDataManipulationException e) {
231 				MessageUtils.addMessageToContext(EXPERIMENT_STORAGE_ACCESS_PROBLEM);
232 
233 				logger.error("A problem accesing storage was encountered when trying to load experiment "
234 						+ experimentName + "  in this backing bean", e);
235 			}
236 		}
237 	}

250 			ExperimentDataManipulationException {
251

253

255 
256 		ExperimentCRUDService crudService = environment.getServiceLocator().getExperimentCRUDService();
257 
258 		this.innerExperiment = crudService.readExperiment(experimentName);
259 
260 		logger.debug("Loaded experiment " + experimentName);
261 
262 		this.loadInnerExperimentCases();
263 	}

266 	 * Loads the experiment cases corresponding to the experiment shown by this
267 	 * bean, along with their status
268 	 */
269 	private void loadInnerExperimentCases() {
270 
271 		logger.debug("Loading experiment cases");
272 
273 		// re-create experiment case list
274 		this.experimentCases = new ArrayList<ExperimentCaseListElement>();
275 		for (Iterator iterator = this.innerExperiment.getExperimentCases().iterator(); iterator.hasNext();) {
276 			ExperimentCaseInfo experimentCase = (ExperimentCaseInfo) iterator.next();
277 			this.experimentCases.add(new ExperimentCaseListElement(experimentCase));
278 		}
279 
280 		Collections.sort(this.experimentCases, new Comparator<ExperimentCaseListElement>() {
281 
282 			@Override
283 			public int compare(ExperimentCaseListElement o1, ExperimentCaseListElement o2) {
284 				return (o1.getName().compareToIgnoreCase(o2.getName()));
285 			}
286 
287 		});
288 
289 		logger.debug("Loaded experiment cases");
290 	}

332 		ExperimentExecutionService execService = environment.getServiceLocator().getExperimentExecutionService();

359 	 * Stops the experiment case indicated in the request parameter
360 	 * 
361 	 * @return a String for the navigation rules indicating that the same view
362 	 *         should be displayed
363 	 */
364 	public String stopExperimentCase() {
365 
366 		String experimentCaseName = FacesUtils.getRequestParameter("experimentCaseName");
367 
368 		if (experimentCaseName == null) {
369 			MessageUtils.handleInternalErrorCondition(REQUEST_PARAM_NOT_FOUND, experimentCaseName);
370 		}
371 
372 		logger.debug("Stopping experiment case " + experimentCaseName);
373 
374 		// if the view holds the experiment case in its own unit of work, an
375 		// Illegal Argument Exception should never be thrown
376 		ExperimentCaseInfo experimentCase = this.innerExperiment.getExperimentCase(experimentCaseName);
377 
378 		ExperimentManagerEnvironment environment = ExperimentManagerEnvironment.getInstance();
379 		ExperimentExecutionService execService = environment.getServiceLocator().getExperimentExecutionService();
380 
381 		try {
382 			// need to check if it's executing since the view might have been
383 			// stale
384 			if (execService.isExecuting(experimentCase)) {
385 				execService.stopExecution(this.innerExperiment, experimentCase);
386 			} else {
387 				// log warning, already not executing
388 				logger.warn("The experiment case " + experimentCaseName
389 						+ " was not be stopped, it was already not executing");
390 			}
391 		}
392 		catch (ExperimentExecutionException e) {
393 			MessageUtils.addMessageToContext(CASE_COULD_NOT_BE_STOPPED, experimentCaseName);
394 			logger.warn("The experiment case " + experimentCaseName
395 					+ " could not be stopped, it may have been already stopped concurrently", e);
396 		}
397 		catch (ExperimentQueryException e) {
398 			MessageUtils.addMessageToContext(CASE_COULD_NOT_BE_STOPPED, experimentCaseName);
399 			logger.warn("The experiment case " + experimentCaseName
400 					+ " could not be stopped, its presence in the remote submission system could not be determined", e);
401 		}
402 
403 		// make it reload the experiment on render - will also reload inner
404 		// experiment cases
405 		ActionResponse response = (ActionResponse) FacesUtils.getPortletResponse();
406 		response.setRenderParameter(EXPERIMENT_NAME, this.innerExperiment.getName());
407 		// to be consistent with javascript on page async monitoring must be
408 		// reset to true - resetBean does that so no separate call to do it is
409 		// needed
410 		this.resetBean();
411 
412 		logger.debug("Stopped experiment case " + experimentCaseName);
413 
414 		return STAY_IN_SAME_VIEW;
415 	}

596 		 * @return the URL to download the input from, as a String, or empty in

602 				return this.inputDownloadURL;
603 			}

608 			}
609 			catch (IOException e) {

611 						+ " due to an IO Exception", e);
612 			}

617 		/**

619 		 * case
620 		 * 
621 		 * @return the URL to download the result from, as a String, or empty if
622 		 *         no result is available yet or in case of error
623 		 */
624 		public String getResultDownloadURL() {
625 
626 			if (this.resultDownloadURL != EMPTY_TEXT_FIELD) {
627 				return this.resultDownloadURL;
628 			}
629 
630 			ExperimentCaseResult expCaseResult;
631 
632 			ExperimentManagerEnvironment environment = ExperimentManagerEnvironment.getInstance();
633 
634 			ExperimentMonitoringService monitorService = environment.getServiceLocator()
635 					.getExperimentMonitoringService();
636 
637 			try {
638 				expCaseResult = monitorService.getResult(this.experimentCase);
639 				if (expCaseResult != null) {
640 					this.resultDownloadURL = this.buildDownloadLink(expCaseResult.getResultFile());
641 				}
642 			}
643 			catch (ExperimentQueryException e) {
644 				logger.warn("Could not get result from native system for experiment case: "
645 						+ this.experimentCase.getName(), e);
646 
647 			}
648 			catch (ExperimentDataManipulationException e) {
649 				logger.warn("Problem accesing storage when searching for experiment case: "
650 						+ this.experimentCase.getName(), e);
651 			}
652 			catch (ExperimentConcurrentAccessException e) {
653 				logger.warn("Could not get result for experiment: " + this.experimentCase.getName()
654 						+ " because storage was being used, another attempt should be made", e);
655 			}
656 			catch (IOException e) {
657 				logger.warn("Could not get result for experiment case: " + this.experimentCase.getName()
658 						+ " due to an IO Exception", e);
659 			}
660 
661 			return this.resultDownloadURL;
662 		}

664 		/**
665 		 * Build the URL to access to download the specified file
666 		 * 
667 		 * @param fileToDownload
668 		 *            the File to download
669 		 * @return the URL to access in order to download the file, as a String
670 		 * @throws IOException
671 		 *             if there was a problem generating the security token or
672 		 *             getting the path of the file on the server
673 		 */
674 		private String buildDownloadLink(File fileToDownload) throws IOException {
675 			String contextPath = FacesUtils.getPortletRequest().getContextPath();
676 
677 			FileDownloadSecurityToken securityToken = new FileDownloadSecurityToken(fileToDownload);
678 
679 			return (contextPath + "/wexpmanDownload?fileName=" + fileToDownload.getCanonicalPath() + "&securityToken=" + securityToken
680 					.toString());
681 		}

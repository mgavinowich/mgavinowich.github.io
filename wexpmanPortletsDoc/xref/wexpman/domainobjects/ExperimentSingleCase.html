72  	 * Constructs the experiment case
73  	 * 
74  	 * @param name
75  	 *            a String with the name for the case
76  	 * @param inputName
77  	 *            a String with the name for this input
78  	 * @param inputData
79  	 *            an InputStream with input data for the case, this input stream
80  	 *            must <b>not</b> be used or closed by the invoker, it will be
81  	 *            read and closed internally once it is used
82  	 * @param ownerName
83  	 *            a (String) name of the owner of the case
84  	 * @param executionContext
85  	 *            the ExperimentExecutionContext with data for the execution of
86  	 *            this experiment case
87  	 * @param holderDirsProvider
88  	 *            an ExperimentHolderDirectoriesProvider which is the provider
89  	 *            of holder directories for input and output files
90  	 * @throws ExperimentCreationException
91  	 *             if the experiment case could not be constructed
92  	 * @throws ExperimentDuplicationException
93  	 *             if a case by the given name already existed in the parent
94  	 *             experiment location
95  	 */
96  	ExperimentSingleCase(String name, String inputName, InputStream inputData, String ownerName,
97  			ExperimentExecutionContext executionContext, ExperimentHolderDirectoriesProvider holderDirsProvider)
98  			throws ExperimentCreationException, ExperimentDuplicationException {
99  		super(name);
100 
101 		try {
102 			this.input = new ExperimentCaseStoredInput(inputName, inputData, holderDirsProvider
103 					.getInputHolderDirectory(), false);
104 		}
105 		catch (IOException e) {
106 			final String errMsg = "The experiment case (" + name
107 					+ ") could not be constructed because its input could not be created";
108 			getLogger().error(errMsg, e);
109 			throw new ExperimentCreationException(errMsg, e);
110 		}
111 
112 		this.ownerName = ownerName;
113 		this.executionContext = executionContext;
114 		this.holderDirsProvider = holderDirsProvider;
115 
116 		this.nativeJob = buildNativeJob();
117 
118 		this.input.freeResources(true);
119 	}

154 		if (this.isSubmitted()) {
155 			throw new IllegalStateException("Cannot start execution for experiment case (" + this.name
156 					+ "), start execution on this experiment case had already been called");
157 		}
158 
159 		getLogger().debug("Starting execution of experiment case (" + this.name + ")");
160 
161 		try {
162 			ExperimentManagerEnvironment.getInstance().getGatewayLocator().getNativeSystemGateway().submitJob(
163 					this.nativeJob);
164 		}
165 		catch (JobSubmissionException e) {
166 			final String errMsg = "The experiment case (" + this.name + ") could not start executing";
167 			getLogger().error(errMsg, e);
168 			throw new ExperimentExecutionException(errMsg, e);
169 		}
170 
171 		getLogger().debug("Execution of experiment case (" + this.name + ") started");
172 	}

217 	/**
218 	 * Builds the native job for submission to the native (remote) submission
219 	 * system
220 	 * 
221 	 * @return a NativeJob for submission to the native submission system
222 	 */
223 	private NativeJob buildNativeJob() {
224 		List<File> inputFile = new ArrayList<File>();
225 		inputFile.add(this.input.getInputFile()); // Collections.singletonList
226 		// was used before, but Db4o
227 		// seemed not to support it
228 		// appropriately
229 
230 		Map<String, String> argsPlaceholderValues = new HashMap<String, String>();
231 		argsPlaceholderValues.put(ExperimentExecutionArguments.INPUT_FILE_NAME_PLACEHOLDER, this.input.getInputFile()
232 				.getName());
233 
234 		NativeJob nativeJob = ExperimentManagerEnvironment.getInstance().getGatewayLocator().getNativeSystemGateway()
235 				.getNewJob(this.name, this.ownerName, this.executionContext.getJobType(),
236 						this.executionContext.getExecutable(),
237 						this.executionContext.getExecutionArguments().getArguments(argsPlaceholderValues), inputFile,
238 						this.executionContext.getExtraRemoteFiles());
239 
240 		return nativeJob;
241 	}

243 	/**
244 	 * Updates and gets the status of this experiment case
245 	 * 
246 	 * @return the ExperimentCaseStatus for this experiment case, if a result is
247 	 *         already loaded it will be the instance
248 	 *         ExperimentCaseStatus.GOT_RESULT; if the experiment case has no
249 	 *         corresponding job on the native (remote) submission system,
250 	 *         either because it was never sent or it has been removed, the
251 	 *         instance ExperimentCaseStatus.NOT_EXECUTING is returned, in all
252 	 *         other cases an ExperimentCaseStatus based on the status of the
253 	 *         job on the native system is returned
254 	 * @throws ExperimentQueryException
255 	 *             if the status of this experiment case could not be obtained
256 	 * 
257 	 * @see ExperimentCase#getStatus()
258 	 */
259 	@Override
260 	public ExperimentCaseStatus getStatus() throws ExperimentQueryException {
261 
262 		if (!this.isExecuting()) {
263 			try {
264 				// try to get (stored - because it is not executing) result - do
265 				// not pass along hook for this
266 				ExperimentCaseResult storedResult = ExperimentCaseResult.getExperimentCaseResult(this.name,
267 						this.nativeJob, this.holderDirsProvider.getOutputHolderDirectory(), this.holderDirsProvider
268 								.getTemporaryDirectory(), null);
269 
270 				if (storedResult != null) {
271 					return ExperimentCaseStatus.GOT_RESULT;
272 				}
273 			}
274 			catch (ExperimentDataManipulationException e) {
275 				// should never happen
276 				final String errMsg = "Experiment case (" + this.name
277 						+ ") stored result could not be obtained, assuming status is NOT_EXECUTING";
278 				getLogger().error(errMsg, e);
279 			}
280 			catch (ExperimentNotCompletedException e) {
281 				// should never happen
282 				final String errMsg = "Experiment case (" + this.name
283 						+ ") stored result could not be obtained, assuming status is NOT_EXECUTING";
284 				getLogger().error(errMsg, e);
285 			}
286 			catch (IOException e) {
287 				final String errMsg = "Experiment case (" + this.name
288 						+ ") stored result could not be obtained, assuming status is NOT_EXECUTING";
289 				getLogger().error(errMsg, e);
290 			}
291 
292 			return ExperimentCaseStatus.NOT_EXECUTING;
293 		}
294 
295 		try {
296 			String statusString = ExperimentManagerEnvironment.getInstance().getGatewayLocator()
297 					.getNativeSystemGateway().getJobStatusString(this.nativeJob);
298 
299 			getLogger().debug(
300 					"Experiment case (" + this.name
301 							+ ") status was obtained from the native submission system and it is: " + statusString);
302 
303 			return new ExperimentCaseStatus(statusString);
304 		}
305 		catch (JobQueryException e) {
306 			final String errMsg = "Experiment case (" + this.name + ") status could not be obtained";
307 			getLogger().error(errMsg, e);
308 			throw new ExperimentQueryException(errMsg, e);
309 		}
310 	}

313 	 * @see wexpman.domainobjects.ExperimentCase#isResultAvailable()
314 	 */
315 	@Override
316 	public boolean isResultAvailable() throws ExperimentQueryException {
317 		boolean resultAvailable = false;
318 
319 		if (!this.isExecuting()) {
320 
321 			ExperimentCaseStatus caseStatus = this.getStatus();
322 
323 			if (caseStatus == ExperimentCaseStatus.GOT_RESULT) {
324 				resultAvailable = true;
325 			} else {
326 				// not executing and no stored result
327 				resultAvailable = false;
328 			}
329 
330 		} else {
331 			// if it's connected to the native system the status can be queried
332 			// (and it necessarily has to have been submitted)
333 			try {
334 				resultAvailable = ExperimentManagerEnvironment.getInstance().getGatewayLocator()
335 						.getNativeSystemGateway().isJobCompleted(this.nativeJob);
336 			}
337 			catch (JobQueryException e) {
338 				final String errMsg = "Experiment case (" + this.name + ") completion status could not be obtained";
339 				getLogger().error(errMsg, e);
340 				throw new ExperimentQueryException(errMsg, e);
341 			}
342 		}
343 
344 		return resultAvailable;
345 	}

355 	/**
356 	 * Gets the result of this experiment case
357 	 * 
358 	 * @return the ExperimentCaseResult of this experiment case
359 	 * @throws ExperimentQueryException
360 	 *             if the result could not be obtained from the native system
361 	 * @throws ExperimentNotCompletedException
362 	 *             if this method was invoked before the experiment case has
363 	 *             completed execution
364 	 * @throws ExperimentDataManipulationException
365 	 *             if the experiment case result instance to be returned could
366 	 *             not be created due to problem trying to store the result data
367 	 * @throws ExperimentConcurrentAccessException
368 	 *             if result permanent storage was being used by another
369 	 *             instance that refers to this same experiment case, this
370 	 *             exception indicates another attempt should be made to get the
371 	 *             result
372 	 * @throws IllegalStateException
373 	 *             if this case has not been submitted, or if it is not
374 	 *             executing and the result had not been previously loaded or
375 	 *             cached
376 	 * 
377 	 * @see ExperimentCase#getResult()
378 	 */
379 	@Override
380 	public ExperimentCaseResult getResult() throws ExperimentQueryException, ExperimentNotCompletedException,
381 			ExperimentDataManipulationException, ExperimentConcurrentAccessException {
382 
383 		ExperimentCaseResult result = null;
384 
385 		if (!this.isSubmitted()) {
386 			final String errMsg = "Cannot ask for result on an experiment case which execution has never been started";
387 			getLogger().error(errMsg);
388 			throw new IllegalStateException(errMsg);
389 		}
390 
391 		getLogger().debug("Getting experiment case (" + this.name + ") result");
392 
393 		try {
394 			result = ExperimentCaseResult.getExperimentCaseResult(this.name, this.nativeJob, this.holderDirsProvider
395 					.getOutputHolderDirectory(), this.holderDirsProvider.getTemporaryDirectory(), this.executionContext
396 					.getHook());
397 		}
398 		catch (IOException e) {
399 			final String errMsg = "There was a problem storing the result of experiment case " + this.name;
400 			getLogger().error(errMsg, e);
401 			throw new ExperimentDataManipulationException(errMsg, e);
402 		}
403 
404 		if (result == null) {
405 			// result was not in local store and could not be obtained from the
406 			// native system
407 			final String errMsg = "Cannot ask for result on an experiment case which is not executing and has not had its result retrieved";
408 			getLogger().error(errMsg);
409 			throw new IllegalStateException(errMsg);
410 		}
411 
412 		getLogger().debug("Got experiment case (" + this.name + ") result");
413 
414 		return result;
415 	}

425 	/**
426 	 * Permanently stops execution of this experiment case
427 	 * 
428 	 * Execution will not be allowed to be restarted
429 	 * 
430 	 * @throws ExperimentExecutionException
431 	 *             if execution of this experiment case could not be stopped
432 	 * @throws IllegalStateException
433 	 *             if this case is not executing
434 	 * 
435 	 * @see ExperimentCase#stopExecution()
436 	 */
437 	@Override
438 	public void stopExecution() throws ExperimentExecutionException {
439 		try {
440 			if (!this.isExecuting()) {
441 				final String errMsg = "Cannot stop execution on an experiment case which is not executing in the native (remote) submission system";
442 				getLogger().error(errMsg);
443 				throw new IllegalStateException(errMsg);
444 			}
445 		}
446 		catch (ExperimentQueryException e) {
447 			final String errMsg = "Could not stop executing experiment case + " + this.name
448 					+ " because the corresponding job presence on the native system could not be verified";
449 			getLogger().error(errMsg, e);
450 			throw new ExperimentExecutionException(errMsg, e);
451 		}
452 
453 		getLogger().debug("Stopping experiment case (" + this.name + ") execution");
454 
455 		try {
456 			ExperimentManagerEnvironment.getInstance().getGatewayLocator().getNativeSystemGateway().removeJob(
457 					this.nativeJob);
458 
459 			getLogger().debug("Stopped execution of experiment case " + this.name);
460 		}
461 		catch (JobCommandException e) {
462 			final String errMsg = "Could not stop executing experiment case + " + this.name
463 					+ " due to a problem stopping the corresponding job on the native system";
464 			getLogger().error(errMsg, e);
465 			throw new ExperimentExecutionException(errMsg, e);
466 		}
467 		catch (UnsupportedOperationException e) {
468 			// this is not a batched case so if this happens it's an internal,
469 			// unrecoverable error
470 			final String errMsg = "Could not remove job on native submission system because the native system gateway interpreted this case as a batched experiment case, when it is actually a simple experiment case";
471 			getLogger().error(errMsg, e);
472 			throw new ExperimentManagerInternalException(errMsg, e);
473 		}
474 	}

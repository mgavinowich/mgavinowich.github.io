69  	/**
70  	 * Added so that it supports embedded server mode
71  	 * 
72  	 * Gets the object container from the resource bound to the thread, using
73  	 * the object server as key
74  	 * 
75  	 */
76  	private ObjectContainer getObjectContainer() {
77  		if (!TransactionSynchronizationManager.hasResource(objectServer)) {
78  			String errMsg = "An object container, bound to the thread with the object server as key, was expected";
79  			throw new IllegalStateException(errMsg);
80  		}
81  
82  		ObjectContainerHolder clientHolder = (ObjectContainerHolder) TransactionSynchronizationManager
83  				.getResource(getResourceBindingKey());
84  
85  		return clientHolder.getObjectContainer();
86  	}

88  	/**
89  	 * Gets the object which is used to bind the session to the thread
90  	 */
91  	private Object getResourceBindingKey() {
92  		// modification: new method, the concept of a resource binding key
93  		// (regardless of what it really is)
94  		// and the use of the ObjectServer as such is introduced in this
95  		// implementation
96  		// all uses of this method are therefore modifications at the place of
97  		// invocation
98  		return this.objectServer;
99  	}

108 	/**
109 	 * @see org.db4ospring.Db4oOperations#execute(org.db4ospring.Db4oCallback,
110 	 *      boolean)
111 	 */
112 	public Object execute(Db4oCallback callback, boolean exposeNativeContainer) throws DataAccessException {
113 		ObjectContainer cont = getObjectContainer();
114 		try {
115 			ObjectContainer container = (exposeNativeContainer ? cont : createContainerProxy(cont));
116 			Object result = callback.doInDb4o(container);
117 			// check out caching/query support
118 			return result;
119 		}
120 		catch (Db4oException ex) {
121 			throw convertDb4oAccessException(ex);
122 		}
123 		catch (RuntimeException ex) {
124 			RuntimeException convEx = convertDb4oAccessException(ex);
125 			// it's user specific
126 			if (convEx instanceof Db4oSystemException) throw ex;
127 			// it's a converted exception
128 			throw convEx;
129 		}
130 	}

581 	 * Create a close-suppressing proxy for the given object container.
582 	 * 
583 	 * @param container
584 	 *            the Db4o ObjectContainer to create a proxy for
585 	 * @return the ObjectContainer proxy
586 	 * 
587 	 * @see com.db4o.ObjectContainer#close()
588 	 */
589 	protected ObjectContainer createContainerProxy(ObjectContainer container) {
590 		Class intrf = ExtObjectContainer.class;
591 
592 		// the documentation states that the container always implements
593 		// the following interface but we do this here just to make sure
594 
595 		if (container instanceof ExtClient) intrf = ExtClient.class;
596 
597 		return (ObjectContainer) Proxy.newProxyInstance(getClass().getClassLoader(), new Class[] { intrf },
598 				new CloseSuppressingInvocationHandler(container));
599 	}

601 	/**
602 	 * Invocation handler that suppresses close calls on Object Container.
603 	 * 
604 	 * @see com.db4o.ObjectContainer#close()
605 	 */
606 	private class CloseSuppressingInvocationHandler implements InvocationHandler {
607 
608 		private final ObjectContainer target;
609 
610 		public CloseSuppressingInvocationHandler(ObjectContainer target) {
611 			this.target = target;
612 		}
613 
614 		public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
615 			// Invocation on ObjectContainer interface (or vendor-specific
616 			// extension) coming in...
617 
618 			if (method.getName().equals("equals")) {
619 				// Only consider equal when proxies are identical.
620 				return (proxy == args[0] ? Boolean.TRUE : Boolean.FALSE);
621 			} else if (method.getName().equals("hashCode")) {
622 				// Use hashCode of session proxy.
623 				return new Integer(hashCode());
624 			} else if (method.getName().equals("close")) {
625 
626 				// Handle close method: suppress, not valid.
627 				// tell the truth when returning the value.
628 				return Boolean.FALSE;
629 			}
630 
631 			// Invoke method on target Session.
632 			try {
633 				Object retVal = method.invoke(this.target, args);
634 
635 				// Note from springmodules implementation: watch out for Query returned
636 				/*
637 				 * if (retVal instanceof Query) { prepareQuery(((Query)
638 				 * retVal)); }
639 				 */
640 				return retVal;
641 			}
642 			catch (InvocationTargetException ex) {
643 				throw ex.getTargetException();
644 			}
645 		}
646 	}

45  	/**
46  	 * Gets the request parameter identified by the given name and sets it as a
47  	 * render parameter, effectively taking it from the action phase to the
48  	 * render phase
49  	 * 
50  	 * This is very much related to the problem described at
51  	 * http://issues.apache.org/jira/browse/MYFACES-788
52  	 * 
53  	 * In brief, JSF knows about a single request that comprises both request
54  	 * portlets know about (action and render). Because of this, parameters
55  	 * included in the pages (e.g. the jsps) or otherwise in action methods, and
56  	 * which are intended to survive until the render phase do not make it
57  	 * 
58  	 * This method should only be invoked during the process action phase
59  	 * 
60  	 * @author Matias Alberto Gavinowich
61  	 * 
62  	 * @param parameterName
63  	 *            the String name of the request parameter to carry onto the
64  	 *            render phase
65  	 * 
66  	 * @throws IllegalStateException
67  	 *             if the method is invoked during a phase other than the action
68  	 *             phase
69  	 * @throws IllegalArgumentException
70  	 *             if no value for the specified parameter is found among the
71  	 *             request parameters
72  	 */
73  	public static void carryRequestParameterOntoRenderPhase(String parameterName) {
74  
75  		PortletResponse response = FacesUtils.getPortletResponse();
76  
77  		if (!(response instanceof ActionResponse)) {
78  			final String errMsg = "Cannot invoke carry request parameter onto render phase method if the current phase is not process action (i.e. response is not an action response)";
79  			logger.error(errMsg);
80  			throw new IllegalStateException(errMsg);
81  		}
82  
83  		String parameterValue = FacesUtils.getRequestParameter(parameterName);
84  
85  		if (parameterValue != null) {
86  			((ActionResponse) response).setRenderParameter(parameterName, parameterValue);
87  		} else {
88  			final String errMsg = "Parameter " + parameterName
89  					+ " could not be carried onto render phase because it was not found";
90  			logger.error(errMsg);
91  			throw new IllegalArgumentException(errMsg);
92  		}
93  	}

95  	/**
96  	 * Finds the first component in the component tree which starts with the
97  	 * given component, which is of the given class
98  	 * 
99  	 * @author Matias Alberto Gavinowich
100 	 * 
101 	 * @param startingComponent
102 	 *            the component that is the root to start searching, can be
103 	 *            UIViewRoot as obtained by {@link FacesContext#getViewRoot()}
104 	 *            because it extends UIComponent
105 	 * @param type
106 	 *            the Class of the component being sought
107 	 * @return the first component of the given Class that is found in the tree,
108 	 *         <b>null</b> if no component of the given class is found
109 	 */
110 	public static UIComponent findComponentByType(UIComponent startingComponent, Class type) {
111 		UIComponent componentFound = null;
112 
113 		if (startingComponent.getClass().equals(type)) {
114 			componentFound = startingComponent;
115 		} else if (startingComponent.getChildCount() > 0) {
116 			for (Iterator<UIComponent> iterator = startingComponent.getChildren().iterator(); iterator.hasNext();) {
117 				UIComponent childComponent = iterator.next();
118 				componentFound = findComponentByType(childComponent, type);
119 				if (componentFound != null) {
120 					break;
121 				}
122 			}
123 		}
124 
125 		return componentFound;
126 	}

129 	 * Gets the portlet preference associated with the given key (preference
130 	 * name)
131 	 * 
132 	 * Assumes there is only one value for the preference, if there are multiple
133 	 * values it will return the first one
134 	 * 
135 	 * @author Matias Alberto Gavinowich
136 	 * 
137 	 * @param preferenceName
138 	 *            the name of the requested preference, as a String
139 	 * @return the value of the portlet preference requested, as a String
140 	 * @throws IllegalArgumentException
141 	 *             if there is no value for a preference by the given name
142 	 */
143 	public static String getPortletPreference(String preferenceName) {
144 		final PortletRequest request = getPortletRequest();
145 		final String preferenceValue = request.getPreferences().getValue(preferenceName, null);
146 		if (preferenceValue == null) {
147 			final String errMsg = "No portlet preference by the name '" + preferenceName + "' was found.";
148 			logger.error(errMsg);
149 			throw new IllegalArgumentException(errMsg);
150 		}
151 		return preferenceValue;
152 	}

155 	 * Get the user name of the user who is currently logged in.
156 	 * 
157 	 * @return a String with the name of the user, or <b>null</b> if it could
158 	 *         not be determined
159 	 */
160 	public static String getPortalUserName() {
161 		String userName = null;
162 		
163 		PortletRequest request = FacesUtils.getPortletRequest();
164 
165 		Map userInfo = (Map) request.getAttribute(PortletRequest.USER_INFO);
166 		userName = (userInfo != null) ? (String) userInfo.get("user.name") : null;
167 		if (userName == null) {
168 			userName = request.getRemoteUser();
169 		}
170 		// may still be null
171 		return userName;
172 	}

269 	 * Evaluate the integer value of a JSF expression.
270 	 * 
271 	 * @param el
272 	 *            the JSF expression
273 	 * @return the integer value associated with the JSF expression
274 	 */
275 	public static Integer evalInt(String el) {
276 		if (el == null) {
277 			return null;
278 		}
279 
280 		if (UIComponentTag.isValueReference(el)) {
281 			Object value = getElValue(el);
282 
283 			if (value == null) {
284 				return null;
285 			} else if (value instanceof Integer) {
286 				return (Integer) value;
287 			} else {
288 				return new Integer(value.toString());
289 			}
290 		} else {
291 			return new Integer(el);
292 		}
293 	}

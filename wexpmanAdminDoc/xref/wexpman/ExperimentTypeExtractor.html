13  /**
14   * Extractor of experiment type values defined in provided data
15   * 
16   * @author Matias Alberto Gavinowich
17   * 
18   */
19  public class ExperimentTypeExtractor {
20  
21  	/**
22  	 * Constructs the experiment case extractor
23  	 */
24  	ExperimentTypeExtractor() {
25  		// creates a new extractor
26  	}
27  
28  	/**
29  	 * Extracts all the experiment type it finds in the given properties
30  	 * 
31  	 * @param properties
32  	 *            a Properties object to analyze
33  	 * @return a Set with the ExperimentTypeValue corresponding to the
34  	 *         experiment types found
35  	 */
36  	Set<ExperimentTypeValue> extractFromProperties(Properties properties) {
37  
38  		Set<ExperimentTypeValue> typesFound = new HashSet<ExperimentTypeValue>();
39  
40  		for (Iterator iterator = properties.keySet().iterator(); iterator.hasNext();) {
41  			String propKey = (String) iterator.next();
42  
43  			int executableSuffixIndex = propKey.lastIndexOf(EXPERIMENT_TYPE_FRIENDLY_NAME_SUFFIX);
44  
45  			if (executableSuffixIndex >= 0) {
46  				String typeId = propKey.substring(0, executableSuffixIndex);
47  				String uiTitle = properties.getProperty(typeId + EXPERIMENT_TYPE_FRIENDLY_NAME_SUFFIX, "");
48  				String jobType = properties.getProperty(typeId + JOB_TYPE_SUFFIX, "");
49  				String executable = properties.getProperty(typeId + EXECUTABLE_SUFFIX, "");
50  				String arguments = properties.getProperty(typeId + ARGUMENTS_SUFFIX, "");
51  				String extraRemoteFiles = properties.getProperty(typeId + EXTRA_REMOTE_FILES_SUFFIX, "");
52  
53  				typesFound.add(new ExperimentTypeValue(typeId, uiTitle, jobType, executable, arguments, extraRemoteFiles));				
54  			}
55  		}
56  
57  		return typesFound;
58  	}
59  }

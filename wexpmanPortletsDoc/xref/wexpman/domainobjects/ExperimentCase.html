49  	/**
50  	 * Asks if the given experiment case name complies with experiment case
51  	 * naming rules
52  	 * 
53  	 * These method should be used to check the name before externally creating
54  	 * an experiment case
55  	 * 
56  	 * @param proposedName
57  	 *            the String name that the experiment case to create would have
58  	 * @return true if the given name complies with experiment case naming
59  	 *         rules, false otherwise
60  	 */
61  	public static boolean isValidCaseName(String proposedName) {
62  
63  		// rules (and which cases have to abide by them):
64  
65  		// 1) for all experiment cases: as case names will be used as
66  		// part of the names of some
67  		// files, no characters which would be invalid in file names can be
68  		// present in case names
69  
70  		// 2) for externally created experiment cases: there is a string which
71  		// is used to
72  		// concatenate case names so that child cases are named after their
73  		// ancestors, this string cannot be used in provided names to prevent
74  		// the case in which an externally created case could be named the same
75  		// as an internally created (child) case, i.e. if provided names don't
76  		// contain that string, it is impossible that their names will collide
77  		// with internally generated names
78  		// Internally created cases will use this string
79  
80  		// validate rule #1 (none of these: \/:*?<>| for win & unix) regex is:
81  		// [\\/:*?<>|]
82  		String lookFor = "[\\\\/:*?<>|]";
83  
84  		Pattern p = Pattern.compile(lookFor);
85  		Matcher m = p.matcher(proposedName);
86  		if (m.find()) {
87  			// the expression was found
88  			return false;
89  		}
90  
91  		// validate rule #2 (none of these: _ )
92  		if (proposedName.contains(INTERNAL_NAMES_CONCATENATE_STR)) {
93  			return false;
94  		}
95  
96  		return true;
97  	}

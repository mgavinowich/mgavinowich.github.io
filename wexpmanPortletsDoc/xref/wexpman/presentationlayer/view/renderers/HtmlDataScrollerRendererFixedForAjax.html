39   * Note by Matias: This is a bug fix for

41   * actually also fixes https://issues.apache.org/jira/browse/TOMAHAWK-596 and

43   * more info at http://marc.info/?l=myfaces-user&m=116237031123948&w=2 Basically

64      /**
65       * Determine which datascroller navigation option the user chose (if any),
66       * and if they did then queue the appropriate ScrollerActionEvent for
67       * later execution.
68       */
69      public void decode(FacesContext context, UIComponent component)
70      {
71          RendererUtils.checkParamValidity(context, component, HtmlDataScroller.class);
72  
73          Map parameter = context.getExternalContext().getRequestParameterMap();
74          String param = (String) parameter.get(component.getClientId(context));
75          if (param != null && param.length() > 0)
76          {
77              if (param.startsWith(PAGE_NAVIGATION))
78              {
79                  // the user chose a specific page# to jump to
80                  component.queueEvent(new ScrollerActionEvent(component, Integer.parseInt(param
81                                  .substring(PAGE_NAVIGATION.length(), param.length()))));
82              }
83              else
84              {
85                  // the user chose first/last/prev/next/fastrewind/fastforward
86                  component.queueEvent(new ScrollerActionEvent(component, param));
87              }
88          }
89      }

92       * Expose much of the internal state of this component so that UIComponents
93       * nested within this component can very flexibly display the component state
94       * to the user.

96      protected void setVariables(FacesContext facescontext, HtmlDataScroller scroller)
97                      throws IOException
98      {
99          Map requestMap = facescontext.getExternalContext().getRequestMap();
100 
101         String pageCountVar = scroller.getPageCountVar();
102         if (pageCountVar != null)
103         {
104             int pageCount = scroller.getPageCount();
105             requestMap.put(pageCountVar, new Integer(pageCount));
106         }
107         String pageIndexVar = scroller.getPageIndexVar();
108         if (pageIndexVar != null)
109         {
110             int pageIndex = (scroller.getRowCount() > 0)? scroller.getPageIndex() : 0;
111             requestMap.put(pageIndexVar, new Integer(pageIndex));
112         }
113         String rowsCountVar = scroller.getRowsCountVar();
114         if (rowsCountVar != null)
115         {
116             int rowsCount = scroller.getRowCount();
117             requestMap.put(rowsCountVar, new Integer(rowsCount));
118         }
119         String displayedRowsCountVar = scroller.getDisplayedRowsCountVar();
120         if (displayedRowsCountVar != null)
121         {
122             int displayedRowsCount = scroller.getRows();
123             int max = scroller.getRowCount() - scroller.getFirstRow();
124             if (displayedRowsCount > max)
125                 displayedRowsCount = max;
126             requestMap.put(displayedRowsCountVar, new Integer(displayedRowsCount));
127         }
128         String firstRowIndexVar = scroller.getFirstRowIndexVar();
129         if (firstRowIndexVar != null)
130         {
131             int firstRowIndex = (scroller.getRowCount() > 0)? scroller.getFirstRow() + 1 : 0;
132             requestMap.put(firstRowIndexVar, new Integer(firstRowIndex));
133         }
134         String lastRowIndexVar = scroller.getLastRowIndexVar();
135         if (lastRowIndexVar != null)
136         {
137             int lastRowIndex = scroller.getFirstRow() + scroller.getRows();
138             int count = scroller.getRowCount();
139             if (lastRowIndex > count)
140                 lastRowIndex = count;
141             requestMap.put(lastRowIndexVar, new Integer(lastRowIndex));
142         }
143     }

219     protected void renderScroller(FacesContext facesContext, HtmlDataScroller scroller)
220                     throws IOException
221     {
222         ResponseWriter writer = facesContext.getResponseWriter();
223 
224         //mhn
225         if (!scroller.isRenderFacetsIfSinglePage() && scroller.getPageCount() <= 1)
226         {
227           // create some dummy links. Otherwise the paging links are not working in subsequent ajax requests!
228           this.createDummyFacetLinks(facesContext, scroller);
229           this.createDummyPaginatorLinks(facesContext, scroller);
230           return;
231         }
232         if (scroller.getFirst() == null && scroller.getFastRewind() == null
233             && scroller.getPrevious() == null && !scroller.isPaginator()
234             && scroller.getNext() == null && scroller.getFastForward() == null
235             && scroller.getLast() == null)
236             return;
237 
238         writer.startElement("table", scroller);
239         String styleClass = scroller.getStyleClass();
240         if (styleClass != null)
241         {
242             writer.writeAttribute("class", styleClass, null);
243         }
244         String style = scroller.getStyle();
245         if (style != null)
246         {
247             writer.writeAttribute("style", style, null);
248         }
249         writer.startElement("tr", scroller);
250 
251         UIComponent facetComp = scroller.getFirst();
252         if (facetComp != null)
253         {
254             writer.startElement("td", scroller);
255             renderFacet(facesContext, scroller, facetComp, HtmlDataScroller.FACET_FIRST);
256             writer.endElement("td");
257         }
258         facetComp = scroller.getFastRewind();
259         if (facetComp != null)
260         {
261             writer.startElement("td", scroller);
262             renderFacet(facesContext, scroller, facetComp, HtmlDataScroller.FACET_FAST_REWIND);
263             writer.endElement("td");
264         }
265         facetComp = scroller.getPrevious();
266         if (facetComp != null)
267         {
268             writer.startElement("td", scroller);
269             renderFacet(facesContext, scroller, facetComp, HtmlDataScroller.FACET_PREVIOUS);
270             writer.endElement("td");
271         }
272         if (scroller.isPaginator())
273         {
274             writer.startElement("td", scroller);
275             renderPaginator(facesContext, scroller);
276             writer.endElement("td");
277         }
278         facetComp = scroller.getNext();
279         if (facetComp != null)
280         {
281             writer.startElement("td", scroller);
282             renderFacet(facesContext, scroller, facetComp, HtmlDataScroller.FACET_NEXT);
283             writer.endElement("td");
284         }
285         facetComp = scroller.getFastForward();
286         if (facetComp != null)
287         {
288             writer.startElement("td", scroller);
289             renderFacet(facesContext, scroller, facetComp, HtmlDataScroller.FACET_FAST_FORWARD);
290             writer.endElement("td");
291         }
292         facetComp = scroller.getLast();
293         if (facetComp != null)
294         {
295             writer.startElement("td", scroller);
296             renderFacet(facesContext, scroller, facetComp, HtmlDataScroller.FACET_LAST);
297             writer.endElement("td");
298         }
299 
300         writer.endElement("tr");
301         writer.endElement("table");
302     }

319     /**
320      * Create dummy links for all pages from 1 until paginatorMaxPages. 
321      * @param facesContext
322      * @param scroller
323      */
324     private void createDummyPaginatorLinks(FacesContext facesContext, HtmlDataScroller scroller)
325     {
326       int maxPages = scroller.getPaginatorMaxPages();
327       if (maxPages <= 1)
328       {
329         maxPages = 2;
330       }
331 
332        for (int idCounter = 1; idCounter <= maxPages; idCounter++)
333        {
334          getLink(facesContext, scroller, 0, idCounter);
335        }
336     }

338     protected void renderFacet(FacesContext facesContext, HtmlDataScroller scroller,
339                                UIComponent facetComp, String facetName) throws IOException
340     {
341         UIComponent link = getLink(facesContext, scroller, facetName);
342         link.encodeBegin(facesContext);
343         facetComp.encodeBegin(facesContext);
344         if (facetComp.getRendersChildren())
345             facetComp.encodeChildren(facesContext);
346         facetComp.encodeEnd(facesContext);
347         link.encodeEnd(facesContext);
348     }

350     /**
351      * The "paginator" is a sequence of page numbers which the user can click
352      * on to leap directly to a specific page of data.
353      */
354     protected void renderPaginator(FacesContext facesContext, HtmlDataScroller scroller)
355                     throws IOException
356     {
357         ResponseWriter writer = facesContext.getResponseWriter();
358 
359         int maxPages = scroller.getPaginatorMaxPages();
360         if (maxPages <= 1)
361         {
362             maxPages = 2;
363         }
364         int pageCount = scroller.getPageCount();
365         if (pageCount <= 1)
366         {
367             return;
368         }
369         int pageIndex = scroller.getPageIndex();
370         int delta = maxPages / 2;
371 
372         int pages;
373         int start;
374         if (pageCount > maxPages && pageIndex > delta)
375         {
376             pages = maxPages;
377             start = pageIndex - pages / 2 - 1;
378             if (start + pages > pageCount)
379             {
380                 start = pageCount - pages;
381             }
382         }
383         else
384         {
385             pages = pageCount < maxPages ? pageCount : maxPages;
386             start = 0;
387         }
388 
389         writer.startElement("table", scroller);
390 
391         String styleClass = scroller.getPaginatorTableClass();
392         if (styleClass != null)
393         {
394             writer.writeAttribute("class", styleClass, null);
395         }
396         String style = scroller.getPaginatorTableStyle();
397         if (style != null)
398         {
399             writer.writeAttribute("style", style, null);
400         }
401 
402         writer.startElement("tr", scroller);
403 
404         int idCounter = 1;
405         for (int i = start, size = start + pages; i < size; i++, idCounter++)
406         {
407             int pagenr = i + 1;
408             writer.startElement("td", scroller);
409             String cStyleClass;
410             String cStyle;
411             if (pagenr == pageIndex)
412             {
413                 cStyleClass = scroller.getPaginatorActiveColumnClass();
414                 cStyle = scroller.getPaginatorActiveColumnStyle();
415             }
416             else
417             {
418                 cStyleClass = scroller.getPaginatorColumnClass();
419                 cStyle = scroller.getPaginatorColumnStyle();
420             }
421             if (cStyleClass != null)
422             {
423                 writer.writeAttribute("class", cStyleClass, null);
424             }
425             if (cStyle != null)
426             {
427                 writer.writeAttribute("style", cStyle, null);
428             }
429 
430             HtmlCommandLink link = getLink(facesContext, scroller, pagenr, idCounter);
431             link.encodeBegin(facesContext);
432             link.encodeChildren(facesContext);
433             link.encodeEnd(facesContext);
434 
435             writer.endElement("td");
436         }
437         
438         //create dummy links until maxPages is reached. These links should not be encoded!
439         // Otherwise the links are not executed after AJAX refreshs and data model updates that result in more
440         // pages.
441         if (idCounter < maxPages)
442         {
443           for (; idCounter < maxPages; idCounter++)
444           {
445               this.getLink(facesContext, scroller, idCounter, idCounter);
446           }
447         }
448 
449         writer.endElement("tr");
450         writer.endElement("table");
451     }

453     protected HtmlCommandLink getLink(FacesContext facesContext, HtmlDataScroller scroller, int pagenr, int idnr)
454     {
455       String text = Integer.toString(pagenr);
456       String generatedIDnr = HtmlDataScrollerRendererFixedForAjax.PAGE_NAVIGATION + idnr;
457       Application application = facesContext.getApplication();
458 
459       HtmlCommandLink link = (HtmlCommandLink) scroller.findComponent(scroller.getId() + generatedIDnr);
460       if (link == null)
461       {
462         link = (HtmlCommandLink) application.createComponent(HtmlCommandLink.COMPONENT_TYPE);
463 
464         link.setId(scroller.getId() + generatedIDnr);
465         link.setTransient(true);
466 
467         this.addParameterToPagerLink(facesContext, scroller, text, generatedIDnr, link);
468       }
469 
470       return link;
471     }

474     protected void addParameterToPagerLink(FacesContext facesContext, HtmlDataScroller scroller, String text,
475         String id, UIComponentBase link)
476     {
477 
478       UIParameter parameter = (UIParameter) facesContext.getApplication().createComponent(
479           UIParameter.COMPONENT_TYPE);
480       parameter.setId(scroller.getId() + id + "_param");
481       parameter.setTransient(true);
482       parameter.setName(scroller.getClientId(facesContext));
483       // parameter.setValue(id);
484 		parameter.setValue(HtmlDataScrollerRendererFixedForAjax.PAGE_NAVIGATION + text);
485       List children = link.getChildren();
486       children.add(parameter);
487       if (text != null)
488       {
489         HtmlOutputText uiText = (HtmlOutputText) facesContext.getApplication().createComponent(
490             HtmlOutputText.COMPONENT_TYPE);
491         uiText.setTransient(true);
492         uiText.setValue(text);
493         children.add(uiText);
494       }
495       scroller.getChildren().add(link);
496     }

498     protected HtmlCommandLink getLink(FacesContext facesContext, HtmlDataScroller scroller, String facetName)
499     {
500       Application application = facesContext.getApplication();
501 
502       HtmlCommandLink link = (HtmlCommandLink) scroller.findComponent(scroller.getId() + facetName);
503       if (link == null)
504       {
505         link = (HtmlCommandLink) application.createComponent(HtmlCommandLink.COMPONENT_TYPE);
506         
507         link.setId(scroller.getId() + facetName);
508         link.setTransient(true);
509         this.addParameterToFacetLink(facesContext, scroller, facetName, link);
510       }
511       return link;
512     }

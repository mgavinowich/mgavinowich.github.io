106 	 * Constructs the CondorJob with the given parameters
107 	 * 
108 	 * @param friendlyName
109 	 *            a (String) friendly name for the job
110 	 * @param owner
111 	 *            a (String) name of the owner of the job
112 	 * @param universe
113 	 *            the Condor universe to submit the job to, e.g. 'vanilla' or
114 	 *            'standard'
115 	 * @param executable
116 	 *            the (String) path to the executable in the remote system
117 	 * @param arguments
118 	 *            the (String) arguments to pass to the executable
119 	 * @param inputFiles
120 	 *            a List of File to be transferred to the remote system and used
121 	 *            as inputs for the job
122 	 * @param remoteInputFiles
123 	 *            a List of String with the paths of files that exist on the
124 	 *            remote system and that should be considered as extra inputs
125 	 *            for the job
126 	 */
127 	CondorJob(String friendlyName, String owner, String universe, String executable, String arguments,
128 			List<File> inputFiles, List<String> remoteInputFiles) {
129 		this.friendlyName = friendlyName;
130 		this.owner = owner;
131 		try {
132 			this.universe = UniverseType.fromString(universe.toUpperCase());
133 		}
134 		catch (IllegalArgumentException e) {
135 			final String errMsg = "The specified universe for this condor job is not a valid UniverseType";
136 			getLogger().error(errMsg, e);
137 			throw new IllegalArgumentException(errMsg, e);
138 		}
139 		this.executable = executable;
140 		this.arguments = arguments;
141 		this.inputFiles = new ArrayList<String>();
142 
143 		CollectionUtils.collect(inputFiles, new Transformer() {
144 
145 			public String transform(Object element) {
146 				File file = (File) element;
147 
148 				try {
149 					return file.getCanonicalPath();
150 				}
151 				catch (IOException e) {
152 					// best effort
153 					return file.getPath();
154 				}
155 			}
156 
157 		}, this.inputFiles);
158 
159 		this.remoteInputFiles = remoteInputFiles;
160 	}

162 	/**
163 	 * Closes the job spool, freeing remote resources. The files for the job can
164 	 * no longer be retrieved from the remote Condor system after closing the
165 	 * spool
166 	 * 
167 	 * @param schedd
168 	 *            a birdbath.Schedd used to contact the schedd daemon
169 	 * @throws JobCommandException
170 	 *             if the spool cannot be closed
171 	 */
172 	void closeSpool(Schedd schedd) throws JobCommandException {
173 
174 		if (!submitted) {
175 			throw new IllegalStateException("Cannot close unsubmitted job (" + this.friendlyName + ") spool");
176 		}
177 
178 		try {
179 
180 			getLogger().debug("Closing spool for " + this.toString());
181 
182 			Transaction transaction = schedd.createTransaction();
183 			final int transactionTimeout = 30;
184 			transaction.begin(transactionTimeout);
185 			closeSpool(transaction);
186 			transaction.commit();
187 
188 			getLogger().debug("Closed spool for " + this.toString());
189 		}
190 		catch (RemoteException e) {
191 			final String errStr = "Close spool command for the job (" + this.friendlyName
192 					+ ") could not be performed due to a problem with the transaction handling";
193 			getLogger().error(errStr, e);
194 			throw new JobCommandException(errStr, e);
195 		}
196 	}

218 	/**
219 	 * Gets the files produced by the job
220 	 * 
221 	 * @param schedd
222 	 *            a birdbath.Schedd used to contact the schedd daemon
223 	 * @return a List of all the File that comprise the job output, can be an
224 	 *         empty list
225 	 * @throws JobNotCompletedException
226 	 *             if this method is invoked before the job has completed
227 	 *             execution
228 	 * @throws JobQueryException
229 	 *             if the job status or spool contents cannot be queried
230 	 * @throws JobDataRetrievalException
231 	 *             if the job output cannot be retrieved
232 	 */
233 	List<File> getOutput(Schedd schedd, File holderDirectory) throws JobNotCompletedException, JobQueryException,
234 			JobDataRetrievalException {
235 
236 		List<File> retrievedFiles = new ArrayList<File>();
237 
238 		if (!submitted) {
239 			throw new IllegalStateException("Cannot ask for job output on unsubmitted job (" + this.friendlyName + ")");
240 		}
241 
242 		// Birdbath currently provides methods to access spool files in the
243 		// Transaction class only
244 		Transaction transaction = null;
245 		FileInfo[] files = null;
246 
247 		try {
248 			if (!this.getCondorStatus(schedd).equals(CondorJobStatus.COMPLETED)) {
249 				// BirdBath provides no transaction.abort at this time
250 				final String errStr = "Output for the job (" + this.friendlyName
251 						+ ") cannot be requested until the job has completed execution";
252 				getLogger().error(errStr);
253 				throw new JobNotCompletedException(errStr);
254 			}
255 
256 			transaction = schedd.createTransaction();
257 			final int transactionTimeout = 30;
258 			transaction.begin(transactionTimeout);
259 			files = transaction.listSpool(clusterId, jobId);
260 			if (files == null) {
261 				getLogger().warn("No output files for job (" + this.friendlyName + ")");
262 				return Collections.emptyList();
263 			}
264 		}
265 		catch (RemoteException e) {
266 			final String errStr = "Spool information for the job (" + this.friendlyName + ") could not be retrieved";
267 			getLogger().error(errStr, e);
268 			throw new JobQueryException(errStr, e);
269 		}
270 
271 		try {
272 			for (int i = 0; i < files.length; i++) {
273 				final File localFile = new File(holderDirectory.getCanonicalPath() + File.separator
274 						+ files[i].getName());
275 				transaction.getFile(clusterId, jobId, files[i].getName(), ((int) files[i].getSize()), localFile);
276 				getLogger().debug(
277 						"Got file " + files[i].getName() + " for job " + this.friendlyName + " " + this.toString());
278 				retrievedFiles.add(localFile);
279 			}
280 
281 			transaction.commit();
282 		}
283 		catch (RemoteException e) {
284 			final String errStr = "Output files for the job (" + this.friendlyName + ") could not be retrieved";
285 			getLogger().error(errStr, e);
286 			throw new JobDataRetrievalException(errStr, e);
287 		}
288 		catch (FileNotFoundException e) {
289 			final String errStr = "Output files for the job (" + this.friendlyName + ") could not be written locally";
290 			getLogger().error(errStr, e);
291 			throw new JobDataRetrievalException(errStr, e);
292 		}
293 		catch (IOException e) {
294 			final String errStr = "Output files for the job (" + this.friendlyName + ") could not be written locally";
295 			getLogger().error(errStr, e);
296 			throw new JobDataRetrievalException(errStr, e);
297 		}
298 
299 		return retrievedFiles;
300 	}

329 	 * Gets this job status
330 	 * 
331 	 * @param schedd
332 	 *            the Schedd to query for the job status
333 	 * @return the job status as a CondorJobStatus
334 	 * @throws JobQueryException
335 	 *             if the query for the job status was unsuccessful
336 	 */
337 	private CondorJobStatus getCondorStatus(Schedd schedd) throws JobQueryException {
338 
339 		CondorJobStatus condorJobStatus = null;
340 
341 		try {
342 
343 			getLogger().debug("Getting job status code for " + this.toString());
344 
345 			ClassAd ad = new ClassAd(schedd.getJobAd(clusterId, jobId));
346 
347 			int status = Integer.parseInt(ad.get(jobStatusClassAdAttribute));
348 
349 			getLogger().debug("Got job status code: " + Integer.toString(status) + " for " + this.toString());
350 
351 			condorJobStatus = CondorJobStatus.fromCondorStatusCode(status);
352 		}
353 		catch (RemoteException e) {
354 			final String errStr = "Could not query the schedd daemon for the job (" + this.friendlyName + ") status";
355 			getLogger().error(errStr, e);
356 			throw new JobQueryException(errStr, e);
357 		}
358 
359 		return condorJobStatus;
360 	}

397 	 * Submits the job
398 	 * 
399 	 * @param transaction
400 	 *            the birdbath.Transaction inside of which the job will be
401 	 *            submitted
402 	 * @param clusterId
403 	 *            the (int) cluster id this job belongs to
404 	 * @throws JobSubmissionException
405 	 *             if the job could not be submitted
406 	 * 
407 	 */
408 	void submit(Transaction transaction, int clusterId) throws JobSubmissionException {
409 
410 		if (submitted) {
411 			throw new IllegalStateException("Cannot submit already submitted job (" + this.friendlyName + ")");
412 		}
413 
414 		getLogger().debug("Submitting job " + this.friendlyName); // job id and
415 		// cluster id
416 		// not set yet
417 
418 		try {
419 			int jobId = transaction.createJob(clusterId);
420 
421 			getLogger().debug("Created job with id of " + Integer.toString(jobId));
422 
423 			// Files to transfer
424 
425 			List<File> files = new ArrayList<File>();
426 
427 			CollectionUtils.collect(this.inputFiles, new Transformer() {
428 
429 				public File transform(Object element) {
430 					String filepath = (String) element;
431 
432 					return new File(filepath);
433 				}
434 
435 			}, files);
436 
437 			File[] fileArray = files.toArray(new File[0]);
438 
439 			List<String> inputFilenames = new ArrayList<String>();
440 
441 			for (File inFile : files) {
442 				inputFilenames.add(inFile.getName());
443 			}
444 
445 			inputFilenames.addAll(this.remoteInputFiles);
446 
447 			// returns an empty string if list is empty
448 			String inputFilenamesJoined = StringUtils.join(inputFilenames, ",");
449 
450 			getLogger().debug("Added " + Integer.toString(fileArray.length) + " files to send with job");
451 
452 			// Extra attributes for the job
453 			ClassAdStructAttr[] extraAttributes = {
454 					new ClassAdStructAttr("TransferExecutable", ClassAdAttrType.value4, "TRUE"),
455 					new ClassAdStructAttr("Out", ClassAdAttrType.value3, this.friendlyName + ".out"),
456 					new ClassAdStructAttr("Err", ClassAdAttrType.value3, this.friendlyName + ".err"),
457 					new ClassAdStructAttr("TransferInput", ClassAdAttrType.value3, inputFilenamesJoined),
458 					new ClassAdStructAttr("UserLog", ClassAdAttrType.value3, this.friendlyName + ".log") };
459 
460 			// Submit the job.
461 			transaction.submit(clusterId, jobId, this.owner, this.universe, this.executable, this.arguments, "(TRUE)",
462 					extraAttributes, fileArray);
463 
464 			// Keep job information
465 			this.clusterId = clusterId;
466 			this.jobId = jobId;
467 			this.submitted = true;
468 
469 			getLogger().debug(
470 					"Submitted job with cluster id of " + Integer.toString(clusterId) + " and jobId of "
471 							+ Integer.toString(jobId));
472 		}
473 		catch (RemoteException e) {
474 			final String errStr = "The Condor job ("
475 					+ this.friendlyName
476 					+ ") could not be submitted because the job could not be created or the input file(s) could not be transferred to the submission host";
477 			getLogger().error(errStr, e);
478 			throw new JobSubmissionException(errStr, e);
479 		}
480 		catch (FileNotFoundException e) {
481 			final String errStr = "The Condor job (" + this.friendlyName
482 					+ ") could not be submitted because the input file(s) could not be found";
483 			getLogger().error(errStr, e);
484 			throw new JobSubmissionException(errStr, e);
485 		}
486 		catch (IOException e) {
487 			final String errStr = "The Condor job (" + this.friendlyName
488 					+ ") could not be submitted because there was an IO problem accessing the input file(s)";
489 			getLogger().error(errStr, e);
490 			throw new JobSubmissionException(errStr, e);
491 		}
492 
493 	}

527 	 * Asks whether this job is running on the native system (running meaning
528 	 * present in the condor queue)
529 	 * 
530 	 * @param schedd
531 	 *            a birdbath.Schedd used to contact the schedd daemon
532 	 * @return true if the job is present on the native system, false
533 	 *         otherwise
534 	 * @throws JobQueryException
535 	 *             if the remote system could not be queried for the job
536 	 */
537 	boolean isPresentInRemoteSystem(Schedd schedd) throws JobQueryException {
538 
539 		this.getLogger().debug("Looking for " + this.toString() + " in the remote submission system");
540 
541 		if (!submitted) {
542 			this.getLogger().debug("The job " + this.toString() + " was never submitted to remote submission system");
543 			return false;
544 		}
545 
546 		// cluster id is ClusterId
547 		// job id is ProcId
548 		try {
549 			this.getLogger().debug("Getting job ad for cluster: " + this.clusterId + " job: " + this.jobId);
550 			String constraint = "ClusterId==" + Integer.toString(this.clusterId) + " && ProcId=="
551 					+ Integer.toString(this.jobId);
552 			ClassAdStructAttr[][] classAds = schedd.getJobAds(constraint);
553 
554 			// ClusterId, ProcId uniquely identify the job, if more than one is
555 			// found then throw an exception - should never happen
556 			if (classAds.length > 1) {
557 				final String errStr = "More than one job for " + this.toString()
558 						+ " was found in the remote submission system";
559 				getLogger().error(errStr);
560 				throw new JobQueryException(errStr);
561 			}
562 
563 			// if the job is found on the native system, return true
564 			if (classAds.length == 1) {
565 				this.getLogger().debug("Found " + this.toString() + " in the remote submission system");
566 				return true;
567 			}
568 
569 			this.getLogger().debug("Did not find " + this.toString() + " in the remote submission system");
570 			return false;
571 		}
572 		catch (RemoteException e) {
573 			final String errStr = "Could not query the schedd daemon for the job (" + this.friendlyName + ") status";
574 			getLogger().error(errStr, e);
575 			throw new JobQueryException(errStr, e);
576 		}
577 	}

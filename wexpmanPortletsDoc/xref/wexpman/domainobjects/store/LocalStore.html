55  	/**
56  	 * Asks whether the given file or directory exists in the local store
57  	 * 
58  	 * @param file
59  	 *            a File that represents the file or directory to be checked
60  	 * @return true if the given file exists in the local store, false otherwise
61  	 * @throws IOException
62  	 *             if there is an I/O problem when checking for file existence
63  	 */
64  	public boolean isStored(File file) throws IOException {
65  
66  		boolean exists = false;
67  
68  		String lockName = this.getClass().getName() + file.getCanonicalPath();
69  
70  		NamedLockProvider lockProvider = NamedLockProvider.getInstance();
71  		NamedLock lock = null;
72  		lock = lockProvider.getLock(lockName, MILLIS_LOCK_TIMEOUT);
73  		if (lock == null) {
74  			final String errMsg = "Could not get named lock " + lockName;
75  			this.logger.error(errMsg);
76  			// some process has locked this file, this condition should never
77  			// happen and will probably not be fully recovered from
78  			// mask it as an IOException this time
79  			throw new IOException(errMsg);
80  		}
81  
82  		try {
83  			exists = file.exists();
84  		}
85  		finally {
86  			lockProvider.releaseLock(lock);
87  		}
88  
89  		return exists;
90  	}

93  	 * Adds a directory to the local store in a thread-safe manner
94  	 * 
95  	 * All necessary parent directories must exist prior to invoking this method
96  	 * 
97  	 * @param directory
98  	 *            a File that represents the directory to add to the store,
99  	 *            which currently must not exist
100 	 * @return true if the directory was added, false otherwise
101 	 * 
102 	 * @throws IllegalArgumentException
103 	 *             if any of the necessary parent directories does not exist
104 	 *             prior to invoking this method
105 	 */
106 	public boolean addDirectory(File directory) {
107 		boolean added = false;
108 
109 		try {
110 			String lockName = this.getClass().getName() + directory.getCanonicalPath();
111 
112 			NamedLockProvider lockProvider = NamedLockProvider.getInstance();
113 			NamedLock lock = null;
114 			lock = lockProvider.getLock(lockName, MILLIS_LOCK_TIMEOUT);
115 			if (lock == null) {
116 				final String errMsg = "Could not get named lock " + lockName;
117 				this.logger.error(errMsg);
118 			} else {
119 				try {
120 					if (!directory.exists()) {
121 						if (directory.mkdir()) {
122 							added = true;
123 						} else {
124 							final String errMsg = "Specified directory ("
125 									+ directory.getName()
126 									+ ") does not exist and could not be created, some parent directories might be necessary but nonexistent";
127 							this.logger.error(errMsg);
128 							throw new IllegalArgumentException(errMsg);
129 						}
130 					}
131 				}
132 				finally {
133 					lockProvider.releaseLock(lock);
134 				}
135 			}
136 		}
137 		catch (IOException e) {
138 			String errMsg = "Could not add specified directory to local store due to an IOException";
139 			logger.error(errMsg, e);
140 			throw new IllegalArgumentException(errMsg);
141 		}
142 
143 		return added;
144 	}

146 	/**
147 	 * Adds a file to the local store from a stream, in a thread-safe manner
148 	 * 
149 	 * @param file
150 	 *            a File that represents the file to add to the store
151 	 * @param data
152 	 *            an InputStream that contains the data to write
153 	 * @param overwriteIfExists
154 	 *            true if the given stream should overwrite the file if it
155 	 *            exists, false otherwise
156 	 * @return true if the file was added, false otherwise
157 	 * @throws IOException
158 	 *             if an I/O problem occurs
159 	 */
160 	public boolean addFile(File file, InputStream data, boolean overwriteIfExists) throws IOException {
161 		boolean added = false;
162 
163 		String lockName = this.getClass().getName() + file.getCanonicalPath();
164 
165 		NamedLockProvider lockProvider = NamedLockProvider.getInstance();
166 		NamedLock lock = null;
167 		lock = lockProvider.getLock(lockName, MILLIS_LOCK_TIMEOUT);
168 		if (lock == null) {
169 			final String errMsg = "Could not get named lock " + lockName;
170 			this.logger.error(errMsg);
171 		} else {
172 			try {
173 				if (overwriteIfExists || (!file.exists())) {
174 					OutputStream out = null;
175 
176 					try {
177 
178 						if (file.exists()) {
179 							if (!file.delete()) {
180 								this.logger.warn("Could not delete previously existing file");
181 							}
182 						}
183 
184 						if (!file.createNewFile()) {
185 							final String errMsg = "The file " + file.getCanonicalPath()
186 									+ " already existed, data is not being saved to the file";
187 							this.logger.error(errMsg);
188 							throw new IllegalArgumentException(errMsg);
189 						}
190 
191 						out = new BufferedOutputStream(new FileOutputStream(file));
192 
193 						int byteRead = data.read();
194 
195 						while (byteRead != -1) {
196 							out.write(byteRead);
197 							byteRead = data.read();
198 						}
199 
200 						added = true;
201 					}
202 					finally {
203 						try {
204 							data.close();
205 						}
206 						catch (IOException e) {
207 							this.logger.warn("Could not close input stream", e);
208 						}
209 						if (out != null) {
210 							try {
211 								out.flush();
212 								out.close();
213 							}
214 							catch (IOException e) {
215 								this.logger.warn("Could not flush or close output stream", e);
216 							}
217 						}
218 					}
219 				}
220 			}
221 			finally {
222 				lockProvider.releaseLock(lock);
223 			}
224 		}
225 
226 		return added;
227 	}

294 		File tempLocation = File.createTempFile(tempDirIdentifier, "tmp", parentDirectory);
295 		if (!tempLocation.delete()) {
296 			logger.warn("Temporary file could not be deleted in order to create a temporary directory in its location");
297 		}
298 		if (!tempLocation.mkdir()) {
299 			String errMsg = "Temporary directory identified by: " + tempDirIdentifier + " could not be created, path: "
300 					+ tempLocation.getPath() + " parent directory: " + parentDirectory.getPath();
301 			logger.error(errMsg);
302 			return null;
303 		}
304 
305 		return tempLocation;
306 	}
